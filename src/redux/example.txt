// React Query data in Redux

Fetch products with React Query in your component.
When data is loaded, dispatch a Redux action (e.g., getProductsListing) to update the Redux store with the fetched products.

Example usage in a component:

import { useProducts } from "../hooks/useProducts";
import { useDispatch } from "react-redux";
import { getProductsListing } from "../redux/productSlice";
import { useEffect } from "react";

function ProductListing() {
  const { data, isLoading, error } = useProducts();
  const dispatch = useDispatch();

  useEffect(() => {
    if (data) {
      dispatch(getProductsListing(data));
    }
  }, [data, dispatch]);

  // ...render products from Redux or React Query
}
